    function copiarCor(cor) {
      navigator.clipboard.writeText(cor)
        .then(() => {
          // Feedback visual (piscar a cor copiada)
          const elementoCor = document.querySelector(`[onclick="copiarCor('${cor}')"]`);
          elementoCor.style.boxShadow = '0 0 10px 5px #e4e1e2';
          setTimeout(() => {
            elementoCor.style.boxShadow = '0 4px 8px rgba(0,0,0,0.1)';
